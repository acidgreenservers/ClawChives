// JavaScript integration example
class JinaProxyStatus {
    constructor() {
        this.indicator = document.getElementById('proxy-indicator');
        this.statusText = document.getElementById('proxy-status-text');
        this.details = document.getElementById('proxy-details');
        this.panel = document.getElementById('status-panel');
        
        this.checkConnection();
        setInterval(() => this.checkConnection(), 60000); // Check every minute
    }
    
    async checkConnection() {
        this.setLoading();
        
        try {
            const response = await fetch('/api/proxy/r.jina/status');
            const data = await response.json();
            
            if (response.ok) {
                this.setOnline(data);
            } else {
                this.setOffline(data.error);
            }
        } catch (error) {
            this.setOffline(error.message);
        }
    }
    
    setLoading() {
        this.indicator.className = 'proxy-status-indicator status-loading';
        this.statusText.textContent = 'Testing r.jina.ai Connection...';
        this.details.textContent = 'Checking...';
    }
    
    setOnline(data) {
        this.indicator.className = 'proxy-status-indicator status-online';
        this.statusText.textContent = 'r.jina.ai Proxy Online';
        this.details.textContent = `Response time: ${data.responseTime}ms`;
        
        // Update detailed panel
        document.getElementById('connection-status').textContent = 'Online';
        document.getElementById('connection-status').className = 'metric-value success';
        document.getElementById('response-time').textContent = `${data.responseTime}ms`;
        document.getElementById('last-check').textContent = 'Just now';
    }
    
    setOffline(error) {
        this.indicator.className = 'proxy-status-indicator status-offline';
        this.statusText.textContent = 'r.jina.ai Proxy Offline';
        this.details.textContent = error || 'Connection failed';
        
        // Update detailed panel
        document.getElementById('connection-status').textContent = 'Offline';
        document.getElementById('connection-status').className = 'metric-value error';
        document.getElementById('last-check').textContent = 'Just now';
    }
}

// Initialize
